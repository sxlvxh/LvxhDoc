下拉框属性参数说明

  略

参数说明

staticResult静态参数配置
dynamicResult动态参数配置，通过接口查询下拉框中的选择数据
event下拉框事件

参数实例

 {
	"staticResult": [{
			"value": "",
			"label": "全部"
		}
	],
	"dynamicResult": {
		"url": "../platmenu/list.action",
		"postDynamicParam": [],
		"postStaticData": {
			"sort": {
				"field": "id desc"
			}
		},
		"resData": {
			"id": "menuCode",
			"value": "name"
		}
	},
	"event": [{
			"name": "change",
			"func": "changeSelect",
			"changeNodes": ["buttonCode"]
		}
	]
}

全静态数据实例

 {
	"staticResult": [{
			"value": "",
			"label": "全部"
		},{
			"value": "0",
			"label": "是"
		},{
			"value": "1",
			"label": "否"
		}
	]
}

 {

			}
		},

			"value": "dataName"
		}
	}
}

接口说明

按用户查询庭审列表

请求路径 【HTTP+JSON】

  http://192.168.2.132:8380/courtWeb/trialcaselistbyrole/grid.action

请求参数说明

partyCode:用户编码 
autoCode:用户建权编码

请求参数示例

{
	"userCode": "13913017145",
	"autoCode": "123456"
}

返回参数说明

result：结果集合
      "treeList": [],
       "createTime": 开始时间
       "createUserId": 创建人id
       "nofuzzy": 模糊查询
       "id": id
       "isdel": 是否逻辑删除 0否1是
       "negativeFields": [],
       "startDate": 庭审开始时间
       "trialCode": 庭审编号
       "caseSeqCode": 案件序列编号
       "caseInfo": 案件描述
       "roomCode": 庭室编码
       "judgeCode": 法官编码
       "clerkCode": 书记员编码
       "entCode": 企业编码
       "roomId": 庭室id
       "syncNo": 同步编号
       "status": 庭审状态
       "trialParams":庭审参数
       "caseCode": 案件编号
       "caseName": 案件名称
       "judgeCodeName": 法官名陈
       "clerkCodeName": 书记员名称
       "entName": 企业名陈
       "trialStatusName": 庭审状态名称
       "activeEntCode": 当前企业编码
 "pages": 分页查询对象
        "curPage": 1,当前分页
        "pageCount": 1,一共多少分页
        "pageSize": 10,分页大小
        "totalCount": 3,查询结果共多少条
        "startNo": 0 从第几条数据开始查询
    }
